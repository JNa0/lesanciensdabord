
const Mongoose = require("mongoose")

const ObjectId = Schéma.Types.ObjectId

// La clé est { id_offre, id_domaine }
module.exports = Mongoose.model("appartient_a", new Schéma({
	// Référence une offre
	"id_offre": ObjectId,
	// Référence un domaine
	"id_domaine": ObjectId,
}))
