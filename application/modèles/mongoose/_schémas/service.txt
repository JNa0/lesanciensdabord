
const Mongoose = require("mongoose")
const Schéma = Mongoose.Schema
const ObjectId = Schéma.Types.ObjectId

module.exports = Mongoose.model("service", new Schéma({
	"id_service": ObjectId,
	"id_organisation": ObjectId,
	"intitulé_service": String,
	"ville_service": String,
	"CP_service": String,
	"pays_service": String,
}))
